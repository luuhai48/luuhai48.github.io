from datetime import datetime
from dateutil import parser

class Coder:
    name = None
    date_of_birth = 0
    year_of_exp = None
    languages = ()

    @property
    def age(self):
        return datetime.now().year() - self.date_of_birth.year()

me = Coder()
me.name = "Lưu Viết Hải"
me.date_of_birth = parser.parse("Aug 04 1997")
me.year_of_exp = 3
me.languages = ("Javascript", "PHP", "Python", "C#")

print(me.age)
