from datetime import datetime
from dateutil import parser

class Coder:
name = None
country = None
date_of_birth = 0
year_of_exp = None
languages = ()

@property
def age(self):
return datetime.now().year - self.date_of_birth.year

me.name = "Lưu Viết Hải"
me.country = "Việt Nam 🇻🇳"

print(me.age)
